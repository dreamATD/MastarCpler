int [][][] func2(int p) {
    int [][][] res;
    res = new int [p][p][p];
    int i, j, k;
    for (i = 0; i < p; ++i) {
        for (j = 0; j < p; ++j) {
            for (k = 0; k < p; ++k) {
                res[i][j][k] = 1;
            }
        }
    }
    return res;
}
